#include <stdio.h>

// Deklaracja funkcji rekurencyjnej
void wypiszLiczby(int n);

int main() {
    int liczba;

    // Wprowadzenie liczby od użytkownika
    printf("Podaj liczbe calkowita: ");
    scanf("%d", &liczba);

    // Wywołanie funkcji i wypisanie liczb
    printf("Liczby od 1 do %d:\n", liczba);
    wypiszLiczby(liczba);

    return 0;
}

// Definicja funkcji rekurencyjnej do wypisywania liczb od 1 do n
void wypiszLiczby(int n) {
    // Warunek zakończenia rekurencji
    if (n > 0) {
        // Wywołanie rekurencyjne
        wypiszLiczby(n - 1);
        // Wypisanie liczby
        printf("%d ", n);
    }
}
